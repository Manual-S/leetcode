## DFS递归解法

```
class Solution 
{
public:
    int maxAreaOfIsland(vector<vector<int>>& grid) 
    {
        if(grid.empty())
        {
            return 0;
        }
        int row = grid.size();
        int col = grid[0].size();

        int maxans = 0;

        for(int i = 0;i < row;i++)
        {
            for(int j = 0;j < col;j++)
            {
                if(grid[i][j] == 1)
                {
                    maxans = std::max(dfs(grid,i,j),maxans);
                }
            }
        }

        return maxans;
    }

private:
    int dfs(vector<vector<int>>& grid,int curx,int cury)
    {
        if(curx < 0 || cury < 0 || curx >= grid.size() || cury >= grid[0].size() || grid[curx][cury] != 1)
        {
            return 0;
        }

        grid[curx][cury] = 0;  //感染
        int dx[] = {-1,1,0,0};
        int dy[]= {0,0,-1,1};  //上下左右四个方向
        int ans = 1;
        for(int i = 0;i < 4;i++)
        {
            int nextx = curx + dx[i];
            int nexty = cury + dy[i];

            ans = ans + dfs(grid,nextx,nexty);
        }
        return ans;
    }
};
```
